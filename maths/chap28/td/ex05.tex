\part{Exercice 5}

On a \[
	P : \quad ax + by + cz + d = 0 \quad \text{ avec } a, b, c, d \neq 0
.\]

\begin{align*}
	\forall i, \quad ax_i + by_i + cz_i + d = 0 \iff& \begin{pmatrix}
		x_1&y_1&z_1&1\\
		x_2&y_2&z_2&1\\
		\vdots&\vdots&\vdots&\vdots
	\end{pmatrix} \begin{pmatrix}
		a\\b\\c\\d
	\end{pmatrix} = \begin{pmatrix}
		0\\\vdots\\0
	\end{pmatrix}
\end{align*}
