	\forall i \in \left\llbracket 1,n \right\rrbracket, e_{i,i}^2 = e_{i,i}

	(e_{k,\ell} + e_{i,i})^2 &= e_{i,i} + e_{k,\ell} e_{i,i} + e_{k,\ell}^2 + e_{i,i}e_{k,\ell} \\
\end{align*}\vspace{-5mm}

	e_{i,j}e_{k,\ell} &= \begin{cases}

\end{align*}
